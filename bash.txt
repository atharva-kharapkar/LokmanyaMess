// ─── simple data table ────────────────────────────────────────────────────────
function dataTable(headers, rows) {
  const COL_WIDTHS = headers.length === 4 ? [800, 3000, 2200, 1400] :
                     headers.length === 6 ? [600, 2600, 2000, 1800, 1600, 1100] :
                     [800, 3800, 2200, 1400];
  const totalW = COL_WIDTHS.reduce((a, b) => a + b, 0);

  function makeCell(text, isHeader = false, align = AlignmentType.CENTER, isLeft = false) {
    return new TableCell({
      borders: tableBorders,
      shading: isHeader ? { fill: NAVY, type: ShadingType.CLEAR } :
               text === 'Total' || text === 'Grand Total' ? { fill: GREY, type: ShadingType.CLEAR } : undefined,
      width: { size: COL_WIDTHS[0], type: WidthType.DXA }, // overridden per cell below
      margins: { top: 80, bottom: 80, left: 120, right: 120 },
      verticalAlign: VerticalAlign.CENTER,
      children: [new Paragraph({
        alignment: isLeft ? AlignmentType.LEFT : align,
        children: [new TextRun({
          text,
          bold: isHeader || text === 'Total' || text === 'Grand Total',
          size: isHeader ? 20 : 21,
          color: isHeader ? WHITE : BLACK,
          font: 'Arial',
        })],
      })],
    });
  }

  const headerRow = new TableRow({
    tableHeader: true,
    children: headers.map((h, i) => {
      const cell = makeCell(h, true, AlignmentType.CENTER, i === 1);
      cell.options.width = { size: COL_WIDTHS[i], type: WidthType.DXA };
      return cell;
    }),
  });

  const dataRows = rows.map((row, ri) =>
    new TableRow({
      children: row.map((val, ci) => {
        const cell = makeCell(String(val), false, ci === 0 ? AlignmentType.CENTER : AlignmentType.LEFT, ci === 1);
        cell.options.width = { size: COL_WIDTHS[ci], type: WidthType.DXA };
        if (ri % 2 === 0 && val !== 'Total') {
          cell.options.shading = { fill: 'F7FBFF', type: ShadingType.CLEAR };
        }
        return cell;
      }),
    })
  );

  return new Table({
    width: { size: totalW, type: WidthType.DXA },
    columnWidths: COL_WIDTHS,
    rows: [headerRow, ...dataRows],
  });
}

// ─── hypothesis summary table ─────────────────────────────────────────────────
function hypothesisTable(rows) {
  const COL = [600, 3400, 1600, 1600, 1200];
  const total = COL.reduce((a, b) => a + b, 0);
  const headers = ['Sr.', 'Hypothesis', 'Tool', 'Composite Mean', 'Result'];

  function hCell(text) {
    return new TableCell({
      borders: tableBorders,
      shading: { fill: NAVY, type: ShadingType.CLEAR },
      margins: { top: 80, bottom: 80, left: 100, right: 100 },
      children: [new Paragraph({
        alignment: AlignmentType.CENTER,
        children: [new TextRun({ text, bold: true, size: 20, color: WHITE, font: 'Arial' })],
      })],
    });
  }

  function dCell(text, ci, ri) {
    const fill = ri % 2 === 0 ? 'F7FBFF' : WHITE;
    return new TableCell({
      borders: tableBorders,
      shading: { fill, type: ShadingType.CLEAR },
      margins: { top: 80, bottom: 80, left: 100, right: 100 },
      children: [new Paragraph({
        alignment: ci === 0 || ci === 3 || ci === 4 ? AlignmentType.CENTER : AlignmentType.LEFT,
        children: [new TextRun({ text: String(text), size: 20, font: 'Arial', color: BLACK })],
      })],
    });
  }

  return new Table({
    width: { size: total, type: WidthType.DXA },
    columnWidths: COL,
    rows: [
      new TableRow({ tableHeader: true, children: headers.map(h => hCell(h)) }),
      ...rows.map((row, ri) => new TableRow({ children: row.map((v, ci) => dCell(v, ci, ri)) })),
    ],
  });
}
