function todayStr() { return new Date().toISOString().split('T')[0]; }
function isValidDate(dateStr) {
  if (!dateStr || typeof dateStr !== 'string') return false;
  const match = dateStr.match(/^(\d{4})-(\d{2})-(\d{2})$/);
  if (!match) return false;
  const year = parseInt(match[1], 10);
  const month = parseInt(match[2], 10);
  const day = parseInt(match[3], 10);
  if (month < 1 || month > 12) return false;
  if (day < 1 || day > 31) return false;
  const dateObj = new Date(year, month - 1, day);
  return dateObj.getFullYear() === year && dateObj.getMonth() === month - 1 && dateObj.getDate() === day;
}
function ini(n) { if (!n) return '?'; return n.trim().split(' ').slice(0,2).map(w => w[0]?.toUpperCase()||'').join(''); }
function rupee(n) { return 'Rs ' + (Number(n)||0).toLocaleString('en-IN'); }
function computeStatus(joinDate, plan) {
  if (!joinDate) return 'active';
  const d = new Date(joinDate); d.setDate(d.getDate() + (PLAN_DAYS[plan]||30));
  const diff = Math.ceil((d - new Date()) / 86400000);
  if (diff < 0) return 'expired'; if (diff <= 2) return 'expiring'; return 'active';
}
function daysLeft(joinDate, plan) {
  if (!joinDate) return null;
  const d = new Date(joinDate); d.setDate(d.getDate() + (PLAN_DAYS[plan]||30));
  return Math.ceil((d - new Date()) / 86400000);
}
function expiryStr(joinDate, plan) {
  if (!joinDate) return '';
  const d = new Date(joinDate); d.setDate(d.getDate() + (PLAN_DAYS[plan]||30));
  return d.toLocaleDateString('en-IN', { day:'2-digit', month:'short', year:'numeric' });
}
function buildMsg(tpl, c, upi, mess) {
  const t = tpl || 'Hi {name},\n\nYour mess subscription has ended.\nPlan: {plan} | Amount: Rs {amount}\n\nPay via UPI: {upi}\n\nThank you\n{mess}';
  return t.replace(/\{name\}/g, c?.name||'').replace(/\{plan\}/g, c?.plan||'').replace(/\{amount\}/g, c?.amount||'').replace(/\{upi\}/g, upi||'Not set').replace(/\{mess\}/g, mess||'Lokmanya Mess');
}function formatWaPhone(phone) {
  if (!phone) return '';
  const digits = phone.replace(/\D/g, '');
  if (digits.length === 10) return '91' + digits;
  return digits;
}


function StableInput({ label, value, onChangeText, placeholder, keyboardType, multiline }) {
  const [local, setLocal] = React.useState(value || '');
  React.useEffect(() => { setLocal(value || ''); }, [value]);
  const handleChange = (text) => { setLocal(text); onChangeText(text); };
  return (
    <View style={{ marginBottom: 13 }}>
      {label ? <Text style={S.label}>{label}</Text> : null}
      <TextInput
        style={[S.input, multiline && { minHeight: 80, textAlignVertical: 'top' }]}
        value={local}
        onChangeText={handleChange}
        placeholder={placeholder}
        placeholderTextColor={C.t2}
        keyboardType={keyboardType || 'default'}
        multiline={!!multiline}
        autoCorrect={false}
        autoCapitalize="none"
        blurOnSubmit={false}
      />
    </View>
  );
}

            <StableInput label="Phone Number" value={adminForm.phone} onChangeText={v=>setAdminForm(f=>({...f,phone:v}))} placeholder="+91XXXXXXXXXX" keyboardType="phone-pad"/>
            <StableInput label={t.messName} value={adminForm.mess} onChangeText={v=>setAdminForm(f=>({...f,mess:v}))} placeholder="e.g. Lokmanya Mess"/>
            <View style={{flexDirection:'row',gap:8,marginTop:4}}>
              <TouchableOpacity style={[S.btn,{flex:1}]} onPress={()=>setAdminModal(false)} activeOpacity={0.7}><Text style={S.btnT}>{t.cancel}</Text></TouchableOpacity>
              <TouchableOpacity style={[S.btn,S.btnP,{flex:2}]} onPress={saveAdmin} activeOpacity={0.7}><Text style={S.btnTW}>{t.saveProfile}</Text></TouchableOpacity>
            </View>
          </View>
        </KeyboardAvoidingView>
      </Modal>

      <Modal visible={tplModal} animationType="slide" transparent onRequestClose={()=>setTplModal(false)}>
        <KeyboardAvoidingView behavior={Platform.OS==='ios'?'padding':'height'} style={S.ov}>
          <ScrollView style={S.sh} showsVerticalScrollIndicator={false} keyboardShouldPersistTaps="handled">
            <Text style={S.shT}>{t.editTemplate}</Text>
            <View style={{marginBottom:10}}>
              <TextInput style={[S.input,{minHeight:120,textAlignVertical:'top'}]} value={tplInput} onChangeText={setTplInput} multiline numberOfLines={6} placeholderTextColor={C.t2} autoCorrect={false}/>
            </View>
            <Text style={[S.label,{marginBottom:6}]}>Preview:</Text>
            <View style={[S.waPrev,{marginBottom:12}]}><Text style={{fontSize:12,color:'#222',lineHeight:18}}>{buildMsg(tplInput,{name:'[Customer]',plan:'Monthly',amount:'1500'},upiId,admin.mess)}</Text></View>
            <View style={{flexDirection:'row',gap:8,marginBottom:20}}>
              <TouchableOpacity style={[S.btn,{flex:1}]} onPress={()=>setTplModal(false)} activeOpacity={0.7}><Text style={S.btnT}>{t.cancel}</Text></TouchableOpacity>
              <TouchableOpacity style={[S.btn,S.btnP,{flex:2}]} onPress={saveTpl} activeOpacity={0.7}><Text style={S.btnTW}>{t.saveMsg}</Text></TouchableOpacity>
            </View>
          </ScrollView>
        </KeyboardAvoidingView>
      </Modal>

      <Modal visible={upiModal} animationType="slide" transparent onRequestClose={()=>setUpiModal(false)}>
        <KeyboardAvoidingView behavior={Platform.OS==='ios'?'padding':'height'} style={S.ov}>
          <View style={S.sh}>
            <Text style={S.shT}>{t.upiId}</Text>
            <StableInput value={upiInput} onChangeText={setUpiInput} placeholder="e.g. name@paytm"/>
            <View style={{flexDirection:'row',gap:8}}>
              <TouchableOpacity style={[S.btn,{flex:1}]} onPress={()=>setUpiModal(false)} activeOpacity={0.7}><Text style={S.btnT}>{t.cancel}</Text></TouchableOpacity>
              <TouchableOpacity style={[S.btn,S.btnP,{flex:2}]} onPress={saveUPI} activeOpacity={0.7}><Text style={S.btnTW}>Save UPI ID</Text></TouchableOpacity>
            </View>
          </View>
        </KeyboardAvoidingView>
      </Modal>

      <Modal visible={conf.open} animationType="fade" transparent onRequestClose={()=>setConf({open:false})}>
        <View style={S.covOv}>
          <View style={S.cBox}>
            <Text style={S.cTit}>{conf.title}</Text>
            <Text style={S.cSub}>{conf.sub}</Text>
            <View style={{flexDirection:'row',gap:8,width:'100%'}}>
              <TouchableOpacity style={[S.btn,{flex:1}]} onPress={()=>setConf({open:false})} activeOpacity={0.7}><Text style={S.btnT}>{t.cancel}</Text></TouchableOpacity>
              <TouchableOpacity style={[S.btn,S.btnD,{flex:1}]} onPress={conf.onOk} activeOpacity={0.7}><Text style={S.btnTW}>{t.delete}</Text></TouchableOpacity>
            </View>
          </View>
        </View>
      </Modal>
    </SafeAreaView>
  );
}
